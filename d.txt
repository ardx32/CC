1. Currency Conversion (INR to USD)This module applies a direct scalar transformation to convert the numerical input based on a static exchange rate.C#

using System;
namespace Currency
{
    class Program
    {
        static void Main(string[] args)
        {
            Console.WriteLine("--- Currency Conversion Engine ---");
            try
            {
                Console.Write("Enter amount in INR: ");
                double inr = Convert.ToDouble(Console.ReadLine());
                
                double conversionRate = 83.0; // Static oracle value
                double usd = inr / conversionRate;
                
                Console.WriteLine($"Result: {inr} INR = {usd:F2} USD");
            }
            catch (FormatException)
            {
                Console.WriteLine("[Error]: Invalid numerical format.");
            }
            Console.ReadLine();
        }
    }
}

2. Factorial ComputationThis implementation utilizes an iterative algorithm, processing in $\mathcal{O}(n)$ time complexity, to calculate the product of all positive descending integers.C#

using System;
namespace Factorial
{
    class Program
    {
        static void Main(string[] args)
        {
            Console.WriteLine("--- Factorial Computation ---");
            try
            {
                Console.Write("Enter a positive integer: ");
                int n = Convert.ToInt32(Console.ReadLine());
                
                if (n < 0) 
                {
                    Console.WriteLine("[Error]: Factorial is strictly undefined for negative integers.");
                    return;
                }
                
                long fact = 1;
                for (int i = 1; i <= n; i++) 
                {
                    fact *= i;
                }
                
                Console.WriteLine($"Factorial of {n} is: {fact}");
            }
            catch (Exception ex)
            {
                Console.WriteLine($"[System Error]: {ex.Message}");
            }
            Console.ReadLine();
        }
    }
}

3. Prime Number ValidationTo optimize the validation process, this algorithm restricts the loop boundary to the square root of $n$, reducing time complexity from $\mathcal{O}(n)$ to $\mathcal{O}(\sqrt{n})$.C#

using System;
namespace Prime
{
    class Program
    {
        static void Main(string[] args)
        {
            Console.WriteLine("--- Prime Validation Engine ---");
            try
            {
                Console.Write("Enter an integer to evaluate: ");
                int n = Convert.ToInt32(Console.ReadLine());
                
                bool isPrime = true;
                if (n <= 1) isPrime = false;
                else if (n > 2)
                {
                    if (n % 2 == 0) isPrime = false;
                    else
                    {
                        int boundary = (int)Math.Floor(Math.Sqrt(n));
                        for (int i = 3; i <= boundary; i += 2)
                        {
                            if (n % i == 0) { isPrime = false; break; }
                        }
                    }
                }

                Console.WriteLine($"Analysis: {n} is {(isPrime ? "a Prime" : "NOT a Prime")} number.");
            }
            catch (FormatException)
            {
                Console.WriteLine("[Error]: Invalid numerical format.");
            }
            Console.ReadLine();
        }
    }
}

9. Arithmetic Calculator SuiteThis module integrates control-flow logic (switch) to act as an execution router, mapping the chosen operator string to the corresponding mathematical constraint.C#

using System;
namespace Calculator
{
    class Program
    {
        static void Main(string[] args)
        {
            Console.WriteLine("--- Full Arithmetic Suite ---");
            try
            {
                Console.Write("Enter first operand: ");
                double a = Convert.ToDouble(Console.ReadLine());
                
                Console.Write("Enter operator (+, -, *, /): ");
                string op = Console.ReadLine();
                
                Console.Write("Enter second operand: ");
                double b = Convert.ToDouble(Console.ReadLine());

                double result = 0;
                switch (op)
                {
                    case "+": result = a + b; break;
                    case "-": result = a - b; break;
                    case "*": result = a * b; break;
                    case "/":
                        if (b == 0) throw new DivideByZeroException("Mathematical constraint violated: Division by zero.");
                        result = a / b;
                        break;
                    default:
                        Console.WriteLine("Invalid operator identified. Use +, -, *, or /.");
                        return;
                }
                Console.WriteLine($"Computed Result: {result}");
            }
            catch (Exception ex)
            {
                Console.WriteLine($"[System Error]: {ex.Message}");
            }
            Console.ReadLine();
        }
    }
}
