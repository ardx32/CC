import javax.ws.rs.*;
import javax.ws.rs.core.MediaType;
import java.util.HashMap;
import java.util.Map;

@Path("student")
public class StudentResource {

    // Simulating a database using RAM (Static memory)
    private static Map<Integer, String> studentDB = new HashMap<>();
    private static int idCounter = 1;

    // CREATE (POST)
    @POST
    @Produces(MediaType.TEXT_PLAIN)
    public String addStudent(@FormParam("name") String name, 
                             @FormParam("age") String age, 
                             @FormParam("phone") String phone, 
                             @FormParam("subject") String subject) {
        String record = "Name: " + name + ", Age: " + age + ", Phone: " + phone + ", Subject: " + subject;
        studentDB.put(idCounter, record);
        return "Student inserted successfully with ID: " + (idCounter++);
    }

    // READ (GET)
    @GET
    @Produces(MediaType.TEXT_PLAIN)
    public String getAllStudents() {
        if (studentDB.isEmpty()) return "Database is empty.";
        StringBuilder allRecords = new StringBuilder("--- Student Database ---\n");
        for (Map.Entry<Integer, String> entry : studentDB.entrySet()) {
            allRecords.append("ID: ").append(entry.getKey()).append(" -> ").append(entry.getValue()).append("\n");
        }
        return allRecords.toString();
    }

    // UPDATE (PUT)
    @PUT
    @Path("{id}")
    @Produces(MediaType.TEXT_PLAIN)
    public String updateStudent(@PathParam("id") int id, 
                                @FormParam("subject") String newSubject) {
        if (studentDB.containsKey(id)) {
            String oldRecord = studentDB.get(id);
            String newRecord = oldRecord + " [Updated Subject: " + newSubject + "]";
            studentDB.put(id, newRecord);
            return "Student ID " + id + " updated.";
        }
        return "Student ID not found.";
    }

    // DELETE (DELETE)
    @DELETE
    @Path("{id}")
    @Produces(MediaType.TEXT_PLAIN)
    public String deleteStudent(@PathParam("id") int id) {
        if (studentDB.remove(id) != null) {
            return "Student ID " + id + " deleted from database.";
        }
        return "Student ID not found.";
    }
}

import javax.ws.rs.*;
import javax.ws.rs.core.MediaType;
import java.util.HashMap;
import java.util.Map;

@Path("employee")
public class EmployeeResource {

    private static Map<Integer, String> employeeDB = new HashMap<>();
    private static int empIdCounter = 101; // Starting ID at 101 for distinction

    // CREATE (POST)
    @POST
    @Produces(MediaType.TEXT_PLAIN)
    public String addEmployee(@FormParam("name") String name, 
                              @FormParam("dept") String dept, 
                              @FormParam("designation") String designation, 
                              @FormParam("salary") String salary) {
        String record = "Name: " + name + ", Dept: " + dept + ", Role: " + designation + ", Salary: $" + salary;
        employeeDB.put(empIdCounter, record);
        return "Employee hired with ID: " + (empIdCounter++);
    }

    // READ (GET)
    @GET
    @Produces(MediaType.TEXT_PLAIN)
    public String getAllEmployees() {
        if (employeeDB.isEmpty()) return "Employee database is empty.";
        StringBuilder allRecords = new StringBuilder("--- Employee Database ---\n");
        for (Map.Entry<Integer, String> entry : employeeDB.entrySet()) {
            allRecords.append("EMP-ID: ").append(entry.getKey()).append(" -> ").append(entry.getValue()).append("\n");
        }
        return allRecords.toString();
    }

    // UPDATE (PUT)
    @PUT
    @Path("{id}")
    @Produces(MediaType.TEXT_PLAIN)
    public String updateSalary(@PathParam("id") int id, 
                               @FormParam("salary") String newSalary) {
        if (employeeDB.containsKey(id)) {
            String oldRecord = employeeDB.get(id);
            employeeDB.put(id, oldRecord + " [Updated Salary: $" + newSalary + "]");
            return "Employee ID " + id + " salary updated.";
        }
        return "Employee ID not found.";
    }

    // DELETE (DELETE)
    @DELETE
    @Path("{id}")
    @Produces(MediaType.TEXT_PLAIN)
    public String deleteEmployee(@PathParam("id") int id) {
        if (employeeDB.remove(id) != null) {
            return "Employee ID " + id + " removed from system.";
        }
        return "Employee ID not found.";
    }
}

package com.api;

import java.util.Set;
import javax.ws.rs.core.Application;

// This annotation tells the server that all REST APIs start with /webresources
@javax.ws.rs.ApplicationPath("webresources")
public class ApplicationConfig extends Application {

    @Override
    public Set<Class<?>> getClasses() {
        Set<Class<?>> resources = new java.util.HashSet<>();
        addRestResourceClasses(resources);
        return resources;
    }

    // Register your API classes here so the server knows they exist
    private void addRestResourceClasses(Set<Class<?>> resources) {
        resources.add(com.api.StudentResource.class);

}
